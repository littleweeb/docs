248  private static int LevenshteinDistance(string source, string target)
249  {
250  // degenerate cases
251  if (source == target) return 0;
252  if (source.Length == 0) return target.Length;
253  if (target.Length == 0) return source.Length;
254 
255  // create two work vectors of integer distances
256  int[] v0 = new int[target.Length + 1];
257  int[] v1 = new int[target.Length + 1];
258 
259  // initialize v0 (the previous row of distances)
260  // this row is A[0][i]: edit distance for an empty s
261  // the distance is just the number of characters to delete from t
262  for (int i = 0; i < v0.Length; i++)
263  v0[i] = i;
264 
265  for (int i = 0; i < source.Length; i++)
266  {
267  // calculate v1 (current row distances) from the previous row v0
268 
269  // first element of v1 is A[i+1][0]
270  // edit distance is delete (i+1) chars from s to match empty t
271  v1[0] = i + 1;
272 
273  // use formula to fill in the rest of the row
274  for (int j = 0; j < target.Length; j++)
275  {
276  var cost = (source[i] == target[j]) ? 0 : 1;
277  v1[j + 1] = Math.Min(v1[j] + 1, Math.Min(v0[j + 1] + 1, v0[j] + cost));
278  }
279 
280  // copy v1 (current row) to v0 (previous row) for next iteration
281  for (int j = 0; j < v0.Length; j++)
282  v0[j] = v1[j];
283  }
284 
285  return v1[target.Length];
286  }
